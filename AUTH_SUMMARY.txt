1. AUTHENTICATION ARCHITECTURE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

MECHANISM: JWT-based authentication with HTTP-only cookies

Backend Stack:
  • Framework: NestJS (Node.js)
  • Token Type: JWT (HS256 algorithm)
  • Password Hashing: bcrypt (10 rounds)
  • Token Storage: HTTP-only cookie (secure, sameSite)
  • Database: Prisma + PostgreSQL

Frontend Stack:
  • Framework: Next.js (React 19)
  • State Management: Zustand
  • HTTP Client: Axios (with interceptors)
  • Validation: Zod schemas
  • Token Transport: Automatic cookie + localStorage

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
2. LOGIN FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

USER INPUT
    ↓
FRONTEND VALIDATION (Zod)
    ↓
POST /api/auth { email, password }
    ↓ (withCredentials: true)
BACKEND AUTH SERVICE
    ├─ Find user by email
    ├─ Compare password with bcrypt hash
    └─ Generate JWT { user_id, email, role }
    ↓
SET COOKIE (httpOnly, secure, sameSite)
    ↓
RETURN { userId, email, role }
    ↓
FRONTEND STORE
    ├─ Zustand state ✅
    └─ localStorage (⚠️ XSS risk)
    ↓
REDIRECT BY ROLE
    ├─ NURSE → /nurse/dashboard
    └─ FAMILY → /dashboard

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
3. TOKEN CONFIGURATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Expiration:
  • Production:   24 hours   (⚠️ Too long, should be 15-30 min)
  • Development:  1 hour     (✅ Good for testing)

JWT Payload:
  {
    user_id:  "uuid",
    email:    "user@example.com",
    role:     "FAMILY" | "NURSE" | "ADMIN",
    iat:      1234567890,      (issued at)
    exp:      1234567890       (expiration)
  }

Signing:
  • Algorithm: HS256 (HMAC SHA-256)
  • Secret: JWT_SECRET from environment
  • ✅ Proper: Secret stored in .env
  • ⚠️ Issue: Example secret is weak

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
4. TOKEN STORAGE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SECURE STORAGE (HTTP-only Cookie)
  ✅ Name:          access_token
  ✅ HttpOnly:      true          (JS cannot access - prevents XSS)
  ✅ Secure:        true (prod)   (HTTPS only)
  ✅ SameSite:      Lax           (CSRF protection)
  ✅ Path:          /
  ✅ Sent with:     All requests (withCredentials: true)

ACCESSIBLE STORAGE (localStorage) - XSS RISK
  ⚠️ userRole:      "FAMILY" | "NURSE" | "ADMIN"
  ⚠️ userId:        "uuid"
  ⚠️ email:         "user@example.com"
  ⚠️ Issue:         Vulnerable to XSS attacks
  ⚠️ Recommendation: Remove from localStorage

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
5. PROTECTED ENDPOINT FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

REQUEST
  POST /api/patients { name, age, ... }
  ↓ withCredentials: true
  Cookie: access_token=<JWT>
    ↓
BACKEND RECEIVES
  ↓
@UseGuards(JwtAuthGuard)
  ├─ Extract token from cookie
  ├─ Verify JWT signature
  ├─ Decode payload → request.user
  └─ If valid: continue
    ↓
CONTROLLER/SERVICE
  ├─ Access request.user.role
  ├─ Check role === 'FAMILY' (in service layer ⚠️)
  └─ Execute logic
    ↓
RESPONSE
  200 OK or 4xx error

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
6. ROLE-BASED ACCESS CONTROL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Three Roles Defined:
  • FAMILY:  Patient/elder family member
  • NURSE:   Healthcare provider
  • ADMIN:   System administrator

Current Implementation:
  ✅ JWT includes role
  ✅ Role checked in service layer
  ✅ Different dashboards per role
  ⚠️ NO role-based guards (@Roles decorator)
  ⚠️ NO admin interface visible
  ⚠️ Role enforcement in service layer (not ideal)

Route Structure:
  /dashboard/           → FAMILY user area
    - patients/
    - bookings/
    - nurses/
  
  /nurse/              → NURSE user area
    - dashboard/
    - availability/
    - earnings/
    - sessions/

7. TOKEN REFRESH MECHANISM
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

CURRENT STATUS: ❌ NOT IMPLEMENTED

Current Flow:
  Token expires (24h prod / 1h dev)
    ↓
  Next API call → 401 Unauthorized
    ↓
  Response interceptor catches 401
    ↓
  Clear localStorage
    ↓
  Redirect to /login
    ↓
  User must re-enter credentials

RECOMMENDED IMPLEMENTATION:
  Access Token:   15-30 minutes  (short-lived)
  Refresh Token:  7 days         (long-lived)
  
  When Access Token Expires:
    ↓
  POST /auth/refresh { refreshToken }
    ↓
  Return new access token
    ↓
  Retry original request

8. LOGOUT FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

CURRENT IMPLEMENTATION (Frontend Only):
  useAuthStore.logout()
    ├─ Clear Zustand state
    ├─ Clear localStorage
    └─ Redirect to /login

ISSUES:
  ❌ No backend logout endpoint
  ❌ Cookie NOT cleared explicitly
  ❌ Token remains valid until expiration
  ❌ No session invalidation

RECOMMENDED:
  POST /auth/logout
    ├─ Backend invalidates token (add to blacklist)
    ├─ Clear cookie immediately
    └─ Frontend clears state & redirects
